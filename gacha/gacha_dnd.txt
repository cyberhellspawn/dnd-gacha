<script>

const epicos=[
"Espada del Dragón",
"Armadura Celestial",
"Anillo del Archimago"
];

const raros=[
"Arco Élfico",
"Botas de Velocidad",
"Capa de Sombras",
"Amuleto de Protección"
];

const comunes=[
"Poción de Curación",
"Daga de Aventurero",
"Escudo de Madera",
"Antorcha",
"Cuerda",
"Ración de Viaje"
];

function randomItem(lista){
return lista[Math.floor(Math.random()*lista.length)];
}

function tirar(){

let r=Math.floor(Math.random()*100)+1;

let texto="";
let clase="";

if(r<=2){
texto=randomItem(epicos);
clase="epico";
}
else if(r<=20){
texto=randomItem(raros);
clase="raro";
}
else{
texto=randomItem(comunes);
clase="comun";
}

document.getElementById("resultado").innerHTML=
`<p class="${clase}">${texto}</p>
<p>(roll ${r})</p>`;

}

</script>
